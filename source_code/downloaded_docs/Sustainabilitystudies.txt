     <p>
      <b>
       Sustainability studies
      </b>
      focus on the interdisciplinary perspective of the
      <a href="/wiki/Sustainability" title="Sustainability">
       sustainability
      </a>
      concept. Programs include instruction in
      <a href="/wiki/Sustainable_development" title="Sustainable development">
       sustainable development
      </a>
      ,
      <a href="/wiki/Geography" title="Geography">
       geography
      </a>
      ,
      <a href="/wiki/Environmental_policy" title="Environmental policy">
       environmental policies
      </a>
      ,
      <a href="/wiki/Ethics" title="Ethics">
       ethics
      </a>
      ,
      <a href="/wiki/Ecology" title="Ecology">
       ecology
      </a>
      ,
      <a href="/wiki/Landscape_architecture" title="Landscape architecture">
       landscape architecture
      </a>
      ,
      <a class="mw-redirect" href="/wiki/City_planning" title="City planning">
       city
      </a>
      and
      <a href="/wiki/Regional_planning" title="Regional planning">
       regional planning
      </a>
      ,
      <a href="/wiki/Economics" title="Economics">
       economics
      </a>
      ,
      <a class="mw-redirect" href="/wiki/Natural_resources" title="Natural resources">
       natural resources
      </a>
      ,
      <a href="/wiki/Sociology" title="Sociology">
       sociology
      </a>
      , and
      <a href="/wiki/Anthropology" title="Anthropology">
       anthropology
      </a>
      .
